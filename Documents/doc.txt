Функция menu выводит возможные команды навигации по проекту, переходы реализованы с помощью конструкции switch-case.

int bcrandom(int *a, int *b, int *c, int *d)
{
    srand(time(NULL));
    do
    {
        *a = 0 + rand() % 9;
        *b = 0 + rand() % 9;
        *c = 0 + rand() % 9;
        *d = 0 + rand() % 9;
    } while (*d == *a || *d == *b || *d == *c || *c == *a || *c == *b || *b == *a);

    printf("Rand number %d%d%d%d\n", *a, *b, *c, *d);

    return 0;
}

Функция bcrandom принимает четыре переменные под хранение цифр сгенерированного числа.
Генерируется случайное число по разрядам, не допуская повторения цифр в числе, и сохраняет в переменные a, b, c, d.

int bulls_cows(int *a, int *b, int *c, int *d)
{
    int sc = 0;
    int number = 0, fst, scnd, thrd, frth;
    int bulls, cows;
    char str[100];

    while (number != (*a * 1000 + *b * 100 + *c * 10 + *d))
    {
        fgets(str, sizeof(str), stdin);
        number = safe_input(str);
        if(number > 0)
        {
            printf("\n");
            fst = number / 1000;
            scnd = number / 100 % 10;
            thrd = number / 10 % 10;
            frth = number % 10;

            bulls = bulls_func(a, b, c, d, &fst, &scnd, &thrd, &frth);
            cows = cows_func(a, b, c, d, &fst, &scnd, &thrd, &frth);
            sc++;

            printf("%d bulls\n", bulls);
            printf("%d cows\n", cows);
            printf("\n");
            printf("\n");

            if(bulls == 4)
                break;
            printf("Enter the four-digit celery: = ");
        }
    }

    printf("Поздравляем, вы выиграли!\nКоличество попыток: %d!\n\nВозвращение в меню...\n\n", sc);
    sleep(2);
    menu();

}

Функция bulls_cows принимает четыре переменные в которых хранится сгенерированное число по разрядам.
Функция выполняется, пока введенное пользователем число не равно сгенерированному числу в функции bcrandom.
В ней происходит вызов функции безопасного ввода safe_input, где осуществляются проверки на правильность ввода.

В функцию safe_input подается строка str, в которую вы что-то ввели.
Там происходят проверки: не ввели ли вы символы отличные от цифр, не ввели ли вы одинаковые цифры; и переводится из типа данных char в int.

После выхода из функции безопасного ввода Ваше предполагаемое число делится по разрядам и отправляется в функции bull_func и cows_func, где подсчитывается количество быков и коров.
Функции bulls_func и cows_func принимают разбитое рандомное число(a, b, c, d) и разбитое вводимое пользователем число(fst, scnd, thrd, frth).

int bulls_func(int *a, int *b, int *c, int *d, int *fst, int *scnd, int *thrd, int *frth)
{
    int bulls = 0;

    if (*fst == *a)
        bulls++;
    if (*scnd == *b)
        bulls++;
    if (*thrd == *c)
        bulls++;
    if (*frth == *d)
        bulls++;

    return bulls;
}

int cows_func(int *a, int *b, int *c, int *d, int *fst, int *scnd, int *thrd, int *frth)
{
    int cows = 0;

    if (*fst == *b || *fst == *c || *fst == *d)
        cows++;
    if (*scnd == *a || *scnd == *c || *scnd == *d)
        cows++;
    if (*thrd == *b || *thrd == *a || *thrd == *d)
        cows++;
    if (*frth == *b || *frth == *c || *frth == *a)
        cows++;

    return cows;
}
